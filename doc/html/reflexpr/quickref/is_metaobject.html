is_metaobject type-trait

The std::is_metaobject type-trait is used to distinguish between metaobject types and all other types. 

template <typename T>
struct is_metaobject
 : integral_constant<bool, implementation-defined >
{ };

template <typename T>
constexpr bool is_metaobject_v = is_metaobject<T>::value;

true if T is a Metaobject false otherwise.

Examples 

using namespace std;

assert(!is_metaobject_v<void>);
assert(!is_metaobject_v<int>);
assert(!is_metaobject_v<string>);

assert( is_metaobject_v<reflexpr(std)>);
assert( is_metaobject_v<reflexpr(int)>);
