ObjectSequence

meta::ObjectSequence is a concept for lazy metaobject sequences. 

template <Object T> concept bool ObjectSequence
 =  implementation-defined;

Operations

Operations applicable to metaobject sequences — types conforming to the meta::ObjectSequence concept: 

// get_size
template <ObjectSequence T>
struct get_size
 : integral_constant<unsigned, implementation-defined >
{ };

template <ObjectSequence T>
constexpr auto get_size_v
 = get_size<T>::value;


// get_element
template <ObjectSequence T, unsigned I>
struct get_element {
	using type = Object 
};

template <ObjectSequence T, unsigned I>
using get_element_m
 = typename get_element<T, I>::type;


// unpack_sequence
template <ObjectSequence T, template <class ...> class Tpl>
struct unpack_sequence {
	using type = Tpl<Object ... >;
};

template <ObjectSequence T, template <class ...> class Tpl>
using unpack_sequence_t
 = typename unpack_sequence<T, Tpl>::type;
