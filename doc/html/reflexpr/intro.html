Hello world

The following code shows a trivial, working example of usage of the __reflexpr operator and the __std_meta_get_base_name and __std_meta_get_scope operations. 

#include <__reflexpr__> 
#include <iostream>

namespace Hello {

struct World { };

} // namespace Hello

int main(void)
{
	using namespace std;

	using meta_HW = __reflexpr(Hello::World);

	cout << __meta_get_base_name_v<__meta_get_scope_m<meta_HW>>;
	cout << ", ";
	cout << __meta_get_base_name_v<meta_HW>;
	cout << "." << std::endl;

	return 0;
}
