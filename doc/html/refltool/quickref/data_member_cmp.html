data member comparisons

#include <refltool/data_member_cmp.hpp>

equal

The call operator of equal_data_members can be used to equality-compare the data members of the instances a and b of a class T. 

template <typename T = void>
struct equal_data_members
{
	constexpr bool operator()(const T& a, const T& b) const;
};

template <>
struct equal_data_members<void>
{
	template <typename T>
	constexpr bool operator()(const T& a, const T& b) const;
};

constexpr const equal_data_members<> data_members_equal = {};
