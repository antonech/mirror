De-serialization using RapidJSON

This example shows how to use the from_rapidjson utility to load instances of nearly arbitrary types from JSON with the RapidJSON library.

We need to include the following headers: 

#include <refltool/from_rapidjson.hpp>
#include <refltool/enum_to_string.hpp>
#include <rapidjson/ostreamwrapper.h>
#include <rapidjson/prettywriter.h>
#include <algorithm>
#include <iostream>

We'll demonstrate the functionality on instances of the following types: 

enum prize_category
{
	physics,
	chemistry,
	medicine,
	literature,
	peace
};

enum laureate_kind
{
	individual,
	organization
};

struct prize_laureate
{
	laureate_kind kind;
	std::string name;

};

struct nobel_prize
{
	prize_category category;
	int year;
	std::vector<prize_laureate> laureates;
};

The from_rapidjson function loads a RapidJSON Value hierarchy into the specified instance. 

int main(void)
{
	using namespace refltool;

	const std::string input(...);

	// parse rapidjson document
	rapidjson::Document doc;
	doc.Parse(input.data(), input.size());

	// load from rapidjson representation
	std::vector<nobel_prize> some_nobel_prizes;
	from_rapidjson(doc, some_nobel_prizes);

	auto is_org = [](const prize_laureate& l) {
		return l.kind == organization;
	};
	auto is_ind = [](const prize_laureate& l) {
		return l.kind == individual;
	};

	for(const auto& prize : some_nobel_prizes) {

		auto ind_n = std::count_if(
			prize.laureates.begin(),
			prize.laureates.end(),
			is_ind
		);

		auto org_n = std::count_if(
			prize.laureates.begin(),
			prize.laureates.end(),
			is_org
		);

		std::cout
			<< enum_to_string(prize.category)
			<< " "
			<< prize.year
			<< ": ";

		if(ind_n > 1) {
			std::cout << ind_n << " individuals";
		} else if (ind_n > 0) {
			std::cout << "1 individual";
		}

		if((ind_n > 0) && (org_n > 0)) {
			std::cout << " and ";
		}

		if(org_n > 1) {
			std::cout << org_n << " organizations";
		} else if (org_n > 0) {
			std::cout << "1 organization";
		}
		std::cout << std::endl;
	}

	return 0;
}
