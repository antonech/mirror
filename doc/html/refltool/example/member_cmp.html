Data members comparison

This example shows how to compare all data members of a class and how to use the data member comparators to easily implement comparison operators.

We need to include the following headers: 

#include <refltool/data_member_cmp.hpp>
#include <iostream>
#include <iomanip>

We'll demonstrate the functionality on the following classes: 

struct Point
{
	int x, y, z;
};

struct Triangle
{
	Point a, b, c;
};

We can use the data_members_equal and data_members_less “operations” to implement equality and less-than comparison operators for our Point and Triangle classes. 

// Point
bool operator == (const Point& a, const Point& b)
{ return refltool::data_members_equal(a, b); }

bool operator <  (const Point& a, const Point& b)
{ return refltool::data_members_less(a, b); }

// Triangle
bool operator == (const Triangle& a, const Triangle& b)
{ return refltool::data_members_equal(a, b); }

bool operator <  (const Triangle& a, const Triangle& b)
{ return refltool::data_members_less(a, b); }

Now we can demonstrate their functionality. 

int main(void)
{
	Point i{1,0,0}, j{0,1,0}, k{0,0,1}, l{1,1,1};

	Triangle d{i,j,k};
	Triangle e{j,k,l};
	Triangle f = d;

	std::cout << "d == d : "  << std::boolalpha << (d == d) << std::endl;
	std::cout << "d == e : "  << std::boolalpha << (d == e) << std::endl;
	std::cout << "d == f : "  << std::boolalpha << (d == f) << std::endl;

	std::cout << "d <  d : "  << std::boolalpha << (d <  d) << std::endl;
	std::cout << "d <  e : "  << std::boolalpha << (d <  e) << std::endl;
	std::cout << "e <  f : "  << std::boolalpha << (e <  f) << std::endl;

	return 0;
}
