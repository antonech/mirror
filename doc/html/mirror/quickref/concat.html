operation concat

#include <mirror/concat.hpp>

The concat operation concatenates several separate ranges or strings into one as shown in the following pseudocode. 

using concat<none, none> = none;

template <typename R>
using concat<R> = R;

template <typename R>
using concat<none, R> = R;

template <typename R>
using concat<R, none> = R;

template <typename Char, Char ... C1, Char ... C2>
using concat<basic_string<Char, C1...>, basic_string<Char, C2...>>
 = basic_string<Char, C1..., C2...>;

template <typename ... P1, typename ... P2>
using concat<range<P1...>, range<P2...>> = range<P1..., P2...>;

template <typename R1, typename R2, typename ... Rn>
using concat<R1, R2, Rn...> = concat<concat<R1, R2>, Rn...>;
