operation is_public

#include <mirror/is_public.hpp>

The is_public operation indicates if the base-level entity reflected by the argument has public access, as shown in the following pseudocode. 

template <std::meta::RecordMember M>
using is_public<metaobject<M>>
 = bool_<std::meta::is_public_v<M>>;

template <std::meta::Inheritance I>
using is_public<metaobject<I>>
 = bool_<std::meta::is_public_v<I>>;

template <std::meta::Object O>
using is_public<metaobject<O>> = none;

using is_public<none> = none;
