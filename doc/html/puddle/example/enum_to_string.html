Enum to string

This example shows how to implement an enum-to-string converter.

We need to include the following headers: 

#include <puddle/metaobject_ops.hpp>
#include <puddle/sequence_ops.hpp>
#include <puddle/reflection.hpp>
#include <puddle/int_const.hpp>
#include <puddle/string.hpp>
#include <iostream>
#include <map>

We'll demonstrate the functionality on the following enum: 

enum class E
{
	a, b, c, d, e
};

The enum_to_string template creates a static instance of a std::map mapping from all enumerators of the Enum passed as argument to the names of the enumerators. 

namespace {

using namespace puddle;

template <typename Enum>
class enum_to_string
{
private:
	static void _eat(bool ...) { }

	template <typename ... MEC>
	static auto _do_make_map(MEC ... mec)
	{
		std::map<Enum, std::string_view> res;
		_eat(res.emplace(
				value(get_constant(mec)),
				string_view(get_base_name(mec))
		).second...);
		return res;
	}

	template <typename ... MEC>
	static auto _make_map(mirror::range<MEC...>)
	{
		return _do_make_map(MEC{}...);
	}
public:
	std::string_view operator()(Enum e) const
	{
		auto MECs = unpack(get_enumerators(PUDDLED(Enum)));
		static auto m = _make_map(MECs);
		return m[e];
	}
};

} // namespace

Now let's try to print the enumerator names to standard output: 

int main(void)
{
	enum_to_string<E> ets;

	std::cout << ets(E::a) << std::endl;
	std::cout << ets(E::b) << std::endl;
	std::cout << ets(E::c) << std::endl;
	std::cout << ets(E::d) << std::endl;
	std::cout << ets(E::e) << std::endl;

	return 0;
}

Running the example produces the following output: 

a
b
c
d
e
