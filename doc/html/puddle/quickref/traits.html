Trait operations

#include <puddle/traits.hpp>

The Puddle library turns the Mirror's trait templates into the following constexpr functions: 

template <typename X>
static constexpr inline
mirror::is_int_const<X> is_int_const(X)
noexcept;

template <typename X>
static constexpr inline
mirror::is_string<X> is_string(X)
noexcept;

static constexpr inline
mirror::is_range<X> is_range(X)
noexcept;

noexcept;
