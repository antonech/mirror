Sequence operations

#include <puddle/sequence_ops.hpp>

The metaobject sequence, range and string operations are implemented by the following functions:

operation is_empty 

template <typename X>
static constexpr inline
mirror::is_empty<X> is_empty(X)
noexcept;

operation get_size 

template <typename X>
static constexpr inline
mirror::get_size<X> get_size(X)
noexcept;

template <typename X>
static constexpr inline
mirror::size<X> size(X)
noexcept;

operation get_element 

template <typename X, typename I>
static constexpr inline
mirror::get_element<X, I> get_element(X, I)
noexcept;

operation unpack 

template <typename X, typename = enable_if_opt_metaobject_sequence<X>>
static constexpr inline
mirror::unpack<X> unpack(X)
noexcept;
