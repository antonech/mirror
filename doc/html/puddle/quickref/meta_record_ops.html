noexcept;

template <typename X>

mirror::is_struct<X> is_struct(X)
noexcept;

operation is_union 

template <typename X>

mirror::is_union<X> is_union(X)
noexcept;

operation base_classes 

template <typename X>
static constexpr inline
mirror::get_base_classes<X> get_base_classes(X)
noexcept;

operation member_types 

template <typename X>
static constexpr inline
mirror::get_member_types<X> get_member_types(X)
noexcept;

operation data_members 

template <typename X>
static constexpr inline
mirror::get_data_members<X> get_data_members(X)
